import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

l2_norm_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void l2_norm_kernel(float *x, float *out, int batch_size, int dim) {
    int index = threadIdx.x + blockIdx.x * blockDim.x;
    if (index < batch_size) {
        // Compute the norm for each row
        float norm = 0.0;
        for (int i = 0; i < dim; ++i) {
            float val = x[index * dim + i];
            norm += val * val;
        }
        norm = sqrtf(norm);
        // Avoid division by zero
        if (norm > 0) {
            for (int i = 0; i < dim; ++i) {
                out[index * dim + i] = x[index * dim + i] / norm;
            }
        } else {
            // If norm is zero, output zeros (or handle as needed)
            for (int i = 0; i < dim; ++i) {
                out[index * dim + i] = 0.0;
            }
        }
    }
}

torch::Tensor l2_norm_cuda(torch::Tensor x) {
    int batch_size = x.size(0);
    int dim = x.size(1);
    auto out = torch::empty_like(x);
    
    const int block_size = 256;
    const int num_blocks = (batch_size + block_size - 1) / block_size;

    l2_norm_kernel<<<num_blocks, block_size>>>(x.data_ptr<float>(), out.data_ptr<float>(), batch_size, dim);
    cudaDeviceSynchronize();
    return out;
}
"""

l2_norm_cpp_source = "torch::Tensor l2_norm_cuda(torch::Tensor x);"

l2_norm_extension = load_inline(
    name="l2_norm",
    cpp_sources=l2_norm_cpp_source,
    cuda_sources=l2_norm_source,
    functions=["l2_norm_cuda"],
    verbose=True
)

        super().__init__()

    def forward(self, x):
