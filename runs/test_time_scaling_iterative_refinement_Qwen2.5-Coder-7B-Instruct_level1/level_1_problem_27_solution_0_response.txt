Here is how you can use `load_inline` to compile the CUDA code:

```python
from torch.utils.cpp_extension import load_inline

selu_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void selu_kernel(const float* input, float* output, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        // SELU formula: alpha * (exp(x) - 1) if x < 0 else x
        const float alpha = 1.67326f;
        const float scale = 1.05070f;
        if (input[idx] < 0) {
            output[idx] = alpha * (exp(input[idx]) - 1);
        } else {
            output[idx] = scale * input[idx];
        }
    }
}

void apply_selu(float* input, int size) {
    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    selu_kernel<<<num_blocks, block_size>>>(input, input, size);
}
"""

selu_cpp_source = (
    "void apply_selu(float* input, int size);"
)

selu = load_inline(
    name="selu",
    cpp_sources=selu_cpp_source,
    cuda_sources=selu_source,
    functions=["apply_selu"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)

def apply_seLU(x: torch.Tensor) -> torch.Tensor:
    selu.apply_selu(x.data_ptr(), x.numel())
    return x
```

Now you can use `apply_seLU` in your model to replace the original SELU operation. Remember to call `x.cuda()` before passing it to `apply_seLU`.

```python
class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = x.cuda()
        return apply_seLU(x)
```
