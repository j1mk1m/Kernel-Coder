#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void depthwise_convolution_kernel(const float* input, const float* weight, float* output, int batch_size, int in_channels, int height, int width, int kernel_size, int stride, int padding) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= batch_size * in_channels * (height + 2 * padding) * (width + 2 * padding)) {
        return;
    }

    int b = idx / (in_channels * (height + 2 * padding) * (width + 2 * padding));
    int c = (idx % (in_channels * (height + 2 * padding) * (width + 2 * padding))) / ((height + 2 * padding) * (width + 2 * padding));
    int h_in = (idx % ((height + 2 * padding) * (width + 2 * padding))) / (width + 2 * padding);
    int w_in = idx % (width + 2 * padding);

    int h_out = (h_in - padding) / stride;
    int w_out = (w_in - padding) / stride;

    if (h_out >= 0 && h_out < height && w_out >= 0 && w_out < width) {
        float sum = 0.0f;
        for (int kh = 0; kh < kernel_size; ++kh) {
            for (int kw = 0; kw < kernel_size; ++kw) {
                int h_weight = kh - padding;
                int w_weight = kw - padding;
                sum += input[(b * in_channels * (height + 2 * padding) * (width + 2 * padding)) + (c * (height + 2 * padding) * (width + 2 * padding)) + ((h_in + h_weight) * (width + 2 * padding)) + (w_in + w_weight)] *
                       weight[(c * kernel_size * kernel_size) + (kh * kernel_size) + kw];
            }
        }
        output[(b * out_channels * height * width) + (c * height * width) + (h_out * width) + w_out] = sum;
    }
}

torch::Tensor depthwise_convolution_cuda(torch::Tensor input, torch::Tensor weight, int kernel_size, int stride, int padding) {
    auto batch_size = input.size(0);
    auto in_channels = input.size(1);
    auto height = input.size(2);
    auto width = input.size(3);
    auto out_channels = weight.size(0);

    auto padded_input = torch::pad(input, {padding, padding, padding, padding}, "constant", 0.0f);
    auto output = torch::zeros({batch_size, out_channels, height, width});

    const int block_size = 256;
    const int num_blocks = (batch_size * in_channels * (height + 2 * padding) * (width + 2 * padding) + block_size - 1) / block_size;

    depthwise_convolution_kernel<<<num_blocks, block_size>>>(padded_input.data_ptr<float>(), weight.data_ptr<float>(), output.data_ptr<float>(), batch_size, in_channels, height, width, kernel_size, stride, padding);

    return output;
}

__global__ void pointwise_convolution_kernel(const float* input, const float* weight, float* output, int batch_size, int in_channels, int height, int width, int out_channels) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= batch_size * in_channels * height * width) {
        return;
    }

    int b = idx / (in_channels * height * width);
    int c_in = (idx % (in_channels * height * width)) / (height * width);
    int h = idx / (width * height);
    int w = idx % width;

    float sum = 0.0f;
    for (int c_out = 0; c_out < out_channels; ++c_out) {
        sum += input[(b * in_channels * height * width) + (c_in * height * width) + (h * width) + w] * weight[c_out * in_channels + c_in];
    }
    output[(b * out_channels * height * width) + (c_out * height * width) + (h * width) + w] = sum;
}

torch::Tensor pointwise_convolution_cuda(torch::Tensor input, torch::Tensor weight, int out_channels) {
    auto batch_size = input.size(0);
    auto in_channels = input.size(1);
    auto height = input.size(2);
    auto width = input.size(3);

    auto output = torch::zeros({batch_size, out_channels, height, width});

    const int block_size = 256;
    const int num_blocks = (batch_size * in_channels * height * width + block_size - 1) / block_size;

    pointwise_convolution_kernel<<<num_blocks, block_size>>>(input.data_ptr<float>(), weight.data_ptr<float>(), output.data_ptr<float>(), batch_size, in_channels, height, width, out_channels);

    return output;
}
"""
