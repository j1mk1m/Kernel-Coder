#include <torch/extension.h>
#include <cuda_runtime.h>

#define BLOCK_SIZE 256

__global__ void relu_kernel(const float* input, float* output, int size) {
    extern __shared__ float sdata[];

    // Load data into shared memory
    int tid = threadIdx.x;
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < size) {
        sdata[tid] = input[i];
    } else {
        sdata[tid] = 0.0f;  // Pad with zeros if index is out of bounds
    }

    __syncthreads();

    // Apply ReLU activation in shared memory
    if (i < size) {
        output[i] = fmaxf(sdata[tid], 0.0f);
    }
}

torch::Tensor relu_cuda(torch::Tensor input) {
    auto size = input.numel();
    auto output = torch::zeros_like(input);

    const int block_size = BLOCK_SIZE;
    const int num_blocks = (size + block_size - 1) / block_size;

    relu_kernel<<<num_blocks, block_size, block_size * sizeof(float)>>>(input.data_ptr<float>(), output.data_ptr<float>(), size);

    return output;
}
