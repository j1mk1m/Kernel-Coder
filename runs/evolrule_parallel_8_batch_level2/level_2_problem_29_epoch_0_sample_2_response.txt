Additional info: the mish function is defined as mish(x) = x * tanh(softplus(x)), which is equivalent to x * tanh(ln(1 + exp(x))). To compute this efficiently, you can compute the softplus as: softplus(x) = max(0, x) + log(1 + exp(-abs(x))). The Mish function has been shown to have a gradient of (exp(x)(x (3 - exp(x)) + 2) + 1)/(exp(x) + 1). But you don't need to worry about the gradient, since PyTorch's autograd will handle it.



```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Custom CUDA kernel for fused linear layer + Mish activation
fused_linear_mish_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// Define Mish computation using optimized softplus
__device__ inline float mish_device(float x) {
    float abs_x = fabsf(x);
    float sp = (x > 0) ? x + logf(1 + expf(-abs_x)) : logf(expf(abs_x) + 1);
    return x * tanhf(sp);
}

template <typename scalar_t>
__global__ void fused_linear_mish_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int batch_size,
    int in_features,
    int out_features) {

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= batch_size * out_features) return;

    int batch_idx = idx / out_features;
    int out_idx = idx % out_features;

    scalar_t sum = bias[out_idx];
    for (int i = 0; i < in_features; ++i) {
        sum += input[batch_idx * in_features + i] * weight[out_idx * in_features + i];
    }

    // Apply Mish activation
    scalar_t x = sum;
    float abs_x = fabsf(x);
    float sp = (x > 0) ? x + logf(1 + expf(-abs_x)) : logf(expf(abs_x) + 1);
    output[idx] = x * tanhf(sp);
}

torch::Tensor fused_linear_mish_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias) {

    const int batch_size = input.size(0);
    const int in_features = input.size(1);
    const int out_features = weight.size(0);

    auto output = torch::empty({batch_size, out_features}, input.options());

    const int threads = 256;
    const int elements = batch_size * out_features;
    const int blocks = (elements + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES(input.type(), "fused_linear_mish_cuda", ([&] {
        fused_linear_mish_kernel<scalar_t><<<blocks, threads>>>(
            input.data<scalar_t>(),
            weight.data<scalar_t>(),
            bias.data<scalar_t>(),
            output.data<scalar_t>(),
            batch_size,
            in_features,
            out_features);
    }));

    return output;
}
"""

cpp_source = "torch::Tensor fused_linear_mish_cuda(torch::Tensor input, torch::Tensor weight, torch::Tensor bias);"

# Compile fused kernel
fused_linear_mish = load_inline(
    name="fused_linear_mish",
    cpp_sources=cpp_source,
    cuda_sources=fused_linear_mish_source,
    functions=["fused_linear_mish_cuda"],
    verbose=True,
    extra_cflags=["-D_GLIBCXX_USE_C99_MATH"],
    extra_ldflags=[""]
)

class ModelNew(nn.Module):
    def __init__(self, in_features, out_features):
        super(ModelNew, self).__init__()
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        self.fused_op = fused_linear_mish
        # Initialize weights and bias like PyTorch's Linear
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / math.sqrt(fan_in)
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        # First Mish application is fused with linear
        x = self.fused_op.fused_linear_mish_cuda(x, self.weight, self.bias)
        # Second Mish applied separately using optimized kernel
        return mish_optimized(x)

# Optimized Mish implementation using inline CUDA
mish_optimized_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

template <typename scalar_t>
__global__ void mish_kernel(scalar_t *input, scalar_t *output, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        scalar_t x = input[idx];
        float abs_x = fabsf(x);
        float sp = (x > 0) ? x + logf(1 + expf(-abs_x)) : logf(expf(abs_x) + 1);
        output[idx] = x * tanhf(sp);
    }
}

torch::Tensor mish_cuda(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int elements = input.numel();
    const int threads = 256;
    const int blocks = (elements + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES(input.type(), "mish_cuda", ([&] {
        mish_kernel<scalar_t><<<blocks, threads>>>(input.data<scalar_t>(), output.data<scalar_t>(), elements);
    }));

    return output;
}
"""

mish_cpp_source = "torch::Tensor mish_cuda(torch::Tensor input);"

mish_optimized = load_inline(
    name="mish_optimized",
    cpp_sources=mish_cpp_source,
    cuda_sources=mish_optimized_source,
    functions=["mish_cuda"],
    verbose=True,
    extra_cflags=["-D_GLIBCXX_USE_C99_MATH"],
    extra_ldflags=[""]
)

def mish_optimized(x):
    return mish_optimized.mish_cuda(x)
```
