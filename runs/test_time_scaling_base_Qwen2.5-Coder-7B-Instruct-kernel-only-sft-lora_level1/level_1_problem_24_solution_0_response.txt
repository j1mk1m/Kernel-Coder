    int tid = threadIdx.x;
    int bid = blockIdx.x;
    int idx = bid * blockDim.x + tid;

    float max_val = -INFINITY;
    if (tid < dim) {
        max_val = x[bid * dim + tid];

    if (tid < dim) {
        sum_exp = expf(x[bid * dim + tid] - max_val);

    sum_exp = __shfl_sync(0xFFFFFFFF, sum_exp, 0);

    if (tid == 0) {
        shared[bid] = sum_exp;
    }
    __syncthreads();

        float total_sum_exp = 0.0f;
        for (int i = 0; i < blockDim.x; ++i) {
