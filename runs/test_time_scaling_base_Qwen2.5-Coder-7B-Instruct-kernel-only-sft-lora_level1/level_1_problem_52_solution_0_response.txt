import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for finding the index of the minimum value
argmin_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void argmin_kernel(const float* x, int* out, int size, int dim) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;

    int min_idx = -1;
    float min_val = std::numeric_limits<float>::max();

    for (int i = tid; i < size; i += stride) {
        int idx = i % size;
        if (idx % dim == 0 && x[idx] < min_val) {
            min_val = x[idx];
            min_idx = idx / dim;
        }
    }

    __shared__ int shared_min_idx[256];
    __shared__ float shared_min_val[256];

    shared_min_idx[threadIdx.x] = min_idx;
    shared_min_val[threadIdx.x] = min_val;

    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s /= 2) {
        if (threadIdx.x < s) {
            if (shared_min_val[threadIdx.x] > shared_min_val[threadIdx.x + s]) {
                shared_min_val[threadIdx.x] = shared_min_val[threadIdx.x + s];
                shared_min_idx[threadIdx.x] = shared_min_idx[threadIdx.x + s];
            } else if (shared_min_val[threadIdx.x] == shared_min_val[threadIdx.x + s] && shared_min_idx[threadIdx.x] > shared_min_idx[threadIdx.x + s]) {
                shared_min_idx[threadIdx.x] = shared_min_idx[threadIdx.x + s];
            }
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        out[blockIdx.x] = shared_min_idx[0];
    }
}

torch::Tensor argmin_cuda(torch::Tensor x, int dim) {
    auto size = x.numel() / dim;
    auto out = torch::empty(size, x.options().dtype(torch::kInt32));

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    argmin_kernel<<<num_blocks, block_size>>>(x.data_ptr<float>(), out.data_ptr<int>(), size, dim);

    return out;
}
"""

argmin_cpp_source = (
    "torch::Tensor argmin_cuda(torch::Tensor x, int dim);"
)

# Compile the inline CUDA code for finding the index of the minimum value
argmin = load_inline(
    name="argmin",
    cpp_sources=argmin_cpp_source,
    cuda_sources=argmin_source,
    functions=["argmin_cuda"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)


class ModelNew(nn.Module):
    """
    Optimized model using custom CUDA operators.
    """

    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to perform argmin on.

        Args:
            dim (int): Dimension along which to find the minimum value.
        """
        super(ModelNew, self).__init__()
        self.dim = dim
        self.argmin = argmin

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Finds the index of the minimum value along the specified dimension using custom CUDA operators.

        Args:
            x (torch.Tensor): Input tensor.

        Returns:
            torch.Tensor: Tensor containing the indices of the minimum values along the specified dimension.
        """
        return self.argmin.argmin_cuda(x, self.dim)
