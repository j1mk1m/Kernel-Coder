import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for minimum operation
min_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void min_kernel(const float* a, float* out, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        out[idx] = a[idx];
    }
}

torch::Tensor min_cuda(torch::Tensor a) {
    auto size = a.numel();
    auto out = torch::zeros_like(a);

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    min_kernel<<<num_blocks, block_size>>>(a.data_ptr<float>(), out.data_ptr<float>(), size);

    return out;
}
"""

min_cpp_source = (
    "torch::Tensor min_cuda(torch::Tensor a);"
)

# Compile the inline CUDA code for minimum operation
min_op = load_inline(
    name="min_op",
    cpp_sources=min_cpp_source,
    cuda_sources=min_source,
    functions=["min_cuda"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)

# Define the custom CUDA kernel for tanh function
tanh_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void tanh_kernel(float* data, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        data[idx] = tanhf(data[idx]);
    }
}

void tanh_cuda(torch::Tensor data) {
    auto size = data.numel();

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    tanh_kernel<<<num_blocks, block_size>>>(data.data_ptr<float>(), size);
}
"""

tanh_cpp_source = (
    "void tanh_cuda(torch::Tensor data);"
)

# Compile the inline CUDA code for tanh function
tanh_op = load_inline(
    name="tanh_op",
    cpp_sources=tanh_cpp_source,
    cuda_sources=tanh_source,
    functions=["tanh_cuda"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.min_op = min_op
        self.tanh_op = tanh_op

    def forward(self, x):
        x = self.conv(x)
        x = self.min_op.min_cuda(x)
        self.tanh_op.tanh_cuda(x)
        self.tanh_op.tanh_cuda(x)
        return x

# Example usage
