__global__ void convolution_3d_kernel(const float* input, const float* weight, float* output, int batch_size, int in_channels, int out_channels, int D, int H, int W, int kernel_size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= batch_size * out_channels * D * H * W) {
        return;
    }

    int o_idx = idx / (D * H * W);
    int d_idx = (idx % (D * H * W)) / (H * W);
    int h_idx = ((idx % (D * H * W)) % (H * W)) / W;
    int w_idx = ((idx % (D * H * W)) % (H * W)) % W;

    float sum = 0.0f;
    for (int i = 0; i < kernel_size; ++i) {
        for (int j = 0; j < kernel_size; ++j) {
            for (int k = 0; k < kernel_size; ++k) {
                int ic_idx = (o_idx % out_channels) * kernel_size * kernel_size * kernel_size + i * kernel_size * kernel_size + j * kernel_size + k;
                int ih_idx = d_idx + i;
                int iw_idx = h_idx + j;
                int iiw_idx = w_idx + k;

                if (ih_idx >= 0 && ih_idx < D && iw_idx >= 0 && iw_idx < H && iiw_idx >= 0 && iiw_idx < W) {
                    sum += input[(iiw_idx * H + iw_idx) * D + ih_idx] * weight[ic_idx];
                }
            }
        }
    }

    output[idx] = sum;
}
