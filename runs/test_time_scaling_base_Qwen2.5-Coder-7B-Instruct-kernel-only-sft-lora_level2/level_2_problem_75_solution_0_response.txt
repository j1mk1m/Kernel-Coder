#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void gemm_kernel(const float* a, const float* b, float* c, int m, int n, int k) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < m && col < n) {
        float sum = 0.0f;
        for (int i = 0; i < k; ++i) {
            sum += a[row * k + i] * b[i * n + col];
        }
        c[row * n + col] = sum;
    }
}

torch::Tensor gemm_cuda(torch::Tensor a, torch::Tensor b) {
    auto m = a.size(0);
    auto n = b.size(1);
    auto k = a.size(1);

    const int block_size = 16;
    const int num_blocks_x = (n + block_size - 1) / block_size;
    const int num_blocks_y = (m + block_size - 1) / block_size;

__global__ void group_norm_kernel(const float* x, float* mean, float* var, float* y, int channels, int groups, int height, int width) {

    int group_channels = channels / groups;
    int group_idx = blockIdx.y * blockDim.y + threadIdx.y;
    int group_col = blockIdx.x * blockDim.x + threadIdx.x;

    float sum = 0.0f;
    float sum_sq = 0.0f;
    int start_channel = group_id * group_channels;
    int end_channel = start_channel + group_channels;

    for (int c = start_channel; c < end_channel; ++c) {
        for (int h = 0; h < height; ++h) {
            for (int w = 0; w < width; ++w) {
                int idx = c * height * width + h * width + w;
                sum += x[idx];
                sum_sq += x[idx] * x[idx];
            }
        }
    }

    __syncthreads();

    if (group_idx == 0 && group_col == 0) {
        mean[group_id] = sum / (height * width * group_channels);
        var[group_id] = sum_sq / (height * width * group_channels) - mean[group_id] * mean[group_id];
    }

    __syncthreads();

    for (int c = start_channel; c < end_channel; ++c) {
        for (int h = 0; h < height; ++h) {
            for (int w = 0; w < width; ++w) {
                int idx = c * height * width + h * width + w;
                y[idx] = (x[idx] - mean[group_id]) / sqrt(var[group_id] + 1e-5f);
            }
        }
    }
}
