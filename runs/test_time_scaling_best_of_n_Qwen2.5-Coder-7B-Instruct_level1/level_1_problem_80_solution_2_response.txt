import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for 2D convolution
convolution_2d_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void convolution_2d_kernel(const float* x, const float* weight, float* out, int batch_size, int in_channels, int out_channels, int height_in, int width_in, int height_out, int width_out, int stride, int padding, int dilation) {
    int n = blockIdx.z * blockDim.z + threadIdx.z;
    int c = blockIdx.y * blockDim.y + threadIdx.y;
    int h = blockIdx.x * blockDim.x + threadIdx.x;

    if (n < batch_size && c < out_channels && h < height_out) {
        int w_start = max(0, (h * stride - padding));
        int w_end = min(width_in, (h * stride - padding + kernel_size[0]));
        int w_step = dilation[1];

        for (int i = 0; i < in_channels; ++i) {
            for (int j = w_start; j < w_end; j += w_step) {
                int w_idx = j - (w_start - padding);
                int x_idx = n * in_channels * height_in * width_in + i * height_in * width_in + h * width_in + j;
                int w_idx_weight = c * in_channels * kernel_size[0] * kernel_size[1] + i * kernel_size[0] * kernel_size[1] + (j - (w_start - padding)) * kernel_size[0] + (w_idx - (w_start - padding));

                atomicAdd(&out[n * out_channels * height_out * width_out + c * height_out * width_out + h * width_out + j], x[x_idx] * weight[w_idx_weight]);
            }
        }

        if (bias != nullptr) {
            atomicAdd(&out[n * out_channels * height_out * width_out + c * height_out * width_out + h * width_out + j], bias[c]);
        }
    }
}

torch::Tensor convolution_2d_cuda(torch::Tensor x, torch::Tensor weight, int stride, int padding, int dilation) {
    auto batch_size = x.size(0);
    auto in_channels = x.size(1);
    auto out_channels = weight.size(0);
    auto height_in = x.size(2);
    auto width_in = x.size(3);
    auto height_out = (height_in + 2 * padding - dilation[0] * (kernel_size[0] - 1) - 1) / stride + 1;
    auto width_out = (width_in + 2 * padding - dilation[1] * (kernel_size[1] - 1) - 1) / stride + 1;

    auto out = torch::zeros({batch_size, out_channels, height_out, width_out}, x.options());

    const int block_size = 16;
    const int num_blocks_x = (width_out + block_size - 1) / block_size;
    const int num_blocks_y = (out_channels + block_size - 1) / block_size;
    const int num_blocks_z = (batch_size + block_size - 1) / block_size;

    convolution_2d_kernel<<<num_blocks_x * num_blocks_y * num_blocks_z, block_size * block_size * block_size>>>(x.data_ptr<float>(), weight.data_ptr<float>(), out.data_ptr<float>(), batch_size, in_channels, out_channels, height_in, width_in, height_out, width_out, stride, padding, dilation);

    return out;
}
"""

convolution_2d_cpp_source = (
    "torch::Tensor convolution_2d_cuda(torch::Tensor x, torch::Tensor weight, int stride, int padding, int dilation);"
)
