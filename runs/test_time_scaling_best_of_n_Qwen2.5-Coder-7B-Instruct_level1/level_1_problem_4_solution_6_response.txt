## Solution:

```python
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for matrix-vector multiplication
matmul_vector_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void matmul_vector_kernel(const float* A, const float* B, float* C, int M, int K) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < M && col == 0) {
        float sum = 0.0f;
        for (int k = 0; k < K; ++k) {
            sum += A[row * K + k] * B[k];
        }
        C[row] = sum;
    }
}

torch::Tensor matmul_vector_cuda(torch::Tensor A, torch::Tensor B) {
    auto M = A.size(0);
    auto K = A.size(1);
    auto out = torch::zeros({M}, A.options());

    const int block_size = 16;
    const int grid_x = (K + block_size - 1) / block_size;
    const int grid_y = (M + block_size - 1) / block_size;

    matmul_vector_kernel<<<grid_y, grid_x>>>(A.data_ptr<float>(), B.data_ptr<float>(), out.data_ptr<float>(), M, K);

    return out;
}
"""

matmul_vector_cpp_source = (
    "torch::Tensor matmul_vector_cuda(torch::Tensor A, torch::Tensor B);"
)

# Compile the inline CUDA code for matrix-vector multiplication
matmul_vector = load_inline(
    name="matmul_vector",
    cpp_sources=matmul_vector_cpp_source,
    cuda_sources=matmul_vector_source,
    functions=["matmul_vector_cuda"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.matmul_vector = matmul_vector

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return self.matmul_vector.matmul_vector_cuda(A, B)
```

This optimized version uses a custom CUDA kernel for matrix-vector multiplication, which should provide a significant performance improvement compared to using PyTorch's built-in `torch.matmul` function.
