import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for Layer Normalization
layer_norm_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void layer_norm_kernel(const float* x, float* mean, float* var, float* out, int batch_size, int features, int dim1, int dim2) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < batch_size * features * dim1 * dim2) {
        int i = idx / (dim1 * dim2);
        int j = idx % (dim1 * dim2);
        int k = i / features;
        int l = i % features;
        int m = j / dim2;
        int n = j % dim2;

        float val = x[idx];
        atomicAdd(&mean[l], val);
        atomicAdd(&var[l], val * val);
    }

    __syncthreads();

    if (threadIdx.x == 0) {
        mean[l] /= (dim1 * dim2);
        var[l] /= (dim1 * dim2);
        var[l] -= mean[l] * mean[l];
        var[l] += 1e-5;  // Small constant to avoid division by zero
    }

    __syncthreads();

    if (idx < batch_size * features * dim1 * dim2) {
        int i = idx / (dim1 * dim2);
        int j = idx % (dim1 * dim2);
        int k = i / features;
        int l = i % features;
        int m = j / dim2;
        int n = j % dim2;

        float val = x[idx];
        out[idx] = (val - mean[l]) / sqrt(var[l]);
    }
}

torch::Tensor layer_norm_cuda(torch::Tensor x) {
    auto batch_size = x.size(0);
    auto features = x.size(1);
    auto dim1 = x.size(2);
    auto dim2 = x.size(3);

    auto mean = torch::zeros(features, device=x.device());
    auto var = torch::ones(features, device=x.device());
    auto out = torch::zeros_like(x);

    const int block_size = 256;
    const int num_blocks = (batch_size * features * dim1 * dim2 + block_size - 1) / block_size;

    layer_norm_kernel<<<num_blocks, block_size>>>(x.data_ptr<float>(), mean.data_ptr<float>(), var.data_ptr<float>(), out.data_ptr<float>(), batch_size, features, dim1, dim2);

    return out;
}
"""
