import torch

import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for masked cumulative sum
masked_cumsum_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <atomic>

__global__ void masked_cumsum_kernel(const float* x, const bool* mask, float* out, int batch_size, int size) {
    extern __shared__ float sdata[];

    int tid = threadIdx.x;
    int bid = blockIdx.x;
    int i = bid * blockDim.x + tid;

    // Initialize shared memory
    sdata[tid] = (i < size && mask[i]) ? x[i] : 0.0f;
    __syncthreads();

    // Perform reduction in shared memory
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }

    // Write result back to global memory
    if (tid == 0) {
        atomicAdd(&out[bid], sdata[0]);
    }
}

torch::Tensor masked_cumsum_cuda(torch::Tensor x, torch::Tensor mask) {
    auto batch_size = x.size(0);
    auto size = x.numel() / batch_size;
    auto out = torch::zeros({batch_size}, torch::kFloat32).to(x.device());

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    masked_cumsum_kernel<<<num_blocks, block_size, block_size * sizeof(float)>>>(x.data_ptr<float>(), mask.data_ptr<bool>(), out.data_ptr<float>(), batch_size, size);

    return out;
}
"""

masked_cumsum_cpp_source = (
    "torch::Tensor masked_cumsum_cuda(torch::Tensor x, torch::Tensor mask);"

    def forward(self, x, mask):

        # Apply masked cumulative sum
