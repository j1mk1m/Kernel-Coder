import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Your custom CUDA kernel code here
custom_softplus_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

// Custom implementation of Softplus using CUDA
__global__ void softplus_kernel(const float* x, float* y, int size) {
    // Implement the Softplus function here
    // For example: y[i] = log(exp(x[i]) + 1.0f);
}

torch::Tensor softplus_cuda(torch::Tensor x) {
    auto size = x.numel();
    auto y = torch::empty_like(x);

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    softplus_kernel<<<num_blocks, block_size>>>(x.data_ptr<float>(), y.data_ptr<float>(), size);

    return y;
}
"""

custom_softplus_cpp_source = (
    "torch::Tensor softplus_cuda(torch::Tensor x);"
)

# Compile the inline CUDA code for custom Softplus
custom_softplus = load_inline(
    name="custom_softplus",
    cpp_sources=custom_softplus_cpp_source,
    cuda_sources=custom_softplus_source,
    functions=["softplus_cuda"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
        self.custom_softplus = custom_softplus

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.custom_softplus.softplus_cuda(x)
