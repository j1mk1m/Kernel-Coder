# Define the custom CUDA kernel for matrix multiplication followed by ReLU and division
matmul_relu_divide_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void matmul_relu_divide_kernel(const float* a, const float* b, float* c, int m, int n, int k, float divisor) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < m && col < n) {
        float sum = 0.0f;
        for (int i = 0; i < k; ++i) {
            sum += a[row * k + i] * b[i * n + col];
        }

        c[row * n + col] = fmaxf(sum, 0.0f);
        c[row * n + col] /= divisor;
    }
}

torch::Tensor matmul_relu_divide_cuda(torch::Tensor a, torch::Tensor b, float divisor) {
    auto m = a.size(0);
    auto n = b.size(1);
    auto k = a.size(1);

    auto c = torch::zeros({m, n}, a.options());

    const int block_size = 16;
    const int num_blocks_x = (n + block_size - 1) / block_size;
    const int num_blocks_y = (m + block_size - 1) / block_size;

    matmul_relu_divide_kernel<<<dim3(num_blocks_x, num_blocks_y), dim3(block_size, block_size)>>>(a.data_ptr<float>(), b.data_ptr<float>(), c.data_ptr<float>(), m, n, k, divisor);

    return c;
}
"""
