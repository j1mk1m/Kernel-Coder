        int w_idx = o_idx / (out_channels * depth * height * in_channels * kernel_size * kernel_size * kernel_size);

        float sum = 0.0f;
        for (int i = 0; i < kernel_size; ++i) {
            for (int j = 0; j < kernel_size; ++j) {
                for (int k = 0; k < kernel_size; ++k) {
