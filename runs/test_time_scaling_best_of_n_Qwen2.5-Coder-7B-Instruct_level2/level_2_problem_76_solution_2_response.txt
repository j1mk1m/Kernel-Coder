#include <cuda_runtime.h>

__global__ void fused_gemm_bias_kernel(const float* a, const float* b, const float* bias, float* out, int batch_size, int in_features, int out_features) {

torch::Tensor fused_gemm_bias_cuda(torch::Tensor a, torch::Tensor b, torch::Tensor bias) {
    auto batch_size = a.size(0);
    auto in_features = a.size(1);
