import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

gemm_leakyrelu_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

#define WARP_SIZE 32

__global__ void gemm_leakyrelu_kernel(float* A, float* B, float* C, float* D, int M, int N, int K, float multiplier, float negative_slope) {
    __shared__ float As[WARP_SIZE][WARP_SIZE];
    __shared__ float Bs[WARP_SIZE][WARP_SIZE];

    int m = blockIdx.y * blockDim.y + threadIdx.y;
    int n = blockIdx.x * blockDim.x + threadIdx.x;
    int k = threadIdx.z;

    float c_value = 0.0f;

    while (k < K) {
        int a_row = m;
        int a_col = k;
        int b_row = k;
        int b_col = n;

        if (a_row < M && a_col < N && b_row < K && b_col < N) {
            As[threadIdx.y][threadIdx.x] = A[a_row * N + a_col];
            Bs[threadIdx.y][threadIdx.x] = B[b_row * N + b_col];
        } else {
            As[threadIdx.y][threadIdx.x] = 0.0f;
            Bs[threadIdx.y][threadIdx.x] = 0.0f;
        }

        __syncthreads();

        for (int i = 0; i < WARP_SIZE; ++i) {
            c_value += As[threadIdx.y][i] * Bs[i][threadIdx.x];
        }

        __syncthreads();

        k += blockDim.z;
    }

    C[m * N + n] = c_value * multiplier;

    __syncthreads();

    if (c_value > 0.0f) {
        D[m * N + n] = c_value;
    } else {
        D[m * N + n] = c_value * negative_slope;
    }
}

torch::Tensor gemm_leakyrelu_cuda(torch::Tensor A, torch::Tensor B, float multiplier, float negative_slope) {
    auto M = A.size(0);
    auto N = B.size(1);
    auto K = A.size(1);

    auto C = torch::zeros({M, N}, A.options());
    auto D = torch::zeros({M, N}, A.options());

    dim3 block_size(WARP_SIZE, WARP_SIZE, 1);
    dim3 grid_size((N + block_size.x - 1) / block_size.x, (M + block_size.y - 1) / block_size.y, 1);

    gemm_leakyrelu_kernel<<<grid_size, block_size>>>(A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), D.data_ptr<float>(), M, N, K, multiplier, negative_slope);

    return D;
}
"""

gemm_leakyrelu_cpp_source = (
    "torch::Tensor gemm_leakyrelu_cuda(torch::Tensor A, torch::Tensor B, float multiplier, float negative_slope);"
)

# Compile the inline CUDA code for GEMM followed by LeakyReLU
gemm_leakyrelu = load_inline(
    name="gemm_leakyrelu",
    cpp_sources=gemm_leakyrelu_cpp_source,
    cuda_sources=gemm_leakyrelu_source,
    functions=["gemm_leakyrelu_cuda"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)


class ModelNew(nn.Module):
    def __init__(self, in_features, out_features, multiplier, negative_slope):
        super(ModelNew, self).__init__()
        self.gemm_leakyrelu = gemm_leakyrelu

    def forward(self, x):
        return self.gemm_leakyrelu.gemm_leakyrelu_cuda(x, x.t(), self.multiplier, self.negative_slope)
