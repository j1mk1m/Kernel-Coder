```python
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for matrix multiplication and Swish activation
matmul_swish_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

// Helper function for sigmoid
__device__ float sigmoid(float x) {
    return 1.0f / (1.0f + exp(-x));
}

// Kernel for matrix multiplication followed by Swish activation
__global__ void matmul_swish_kernel(const float* a, const float* b, float* c, int rows_a, int cols_a, int cols_b) {
    int row_idx = blockIdx.y * blockDim.y + threadIdx.y;
    int col_idx = blockIdx.x * blockDim.x + threadIdx.x;

    if (row_idx < rows_a && col_idx < cols_b) {
        float sum = 0.0f;
        for (int k = 0; k < cols_a; ++k) {
            sum += a[row_idx * cols_a + k] * b[k * cols_b + col_idx];
        }
        c[row_idx * cols_b + col_idx] = sum * sigmoid(sum);
    }
}

torch::Tensor matmul_swish_cuda(torch::Tensor a, torch::Tensor b, float scaling_factor) {
    auto rows_a = a.size(0);
    auto cols_a = a.size(1);
    auto cols_b = b.size(1);
    auto out = torch::zeros({rows_a, cols_b}, a.options());

    const int block_size_x = 32;
    const int block_size_y = 32;
    const int num_blocks_x = (cols_b + block_size_x - 1) / block_size_x;
    const int num_blocks_y = (rows_a + block_size_y - 1) / block_size_y;

    matmul_swish_kernel<<<dim3(num_blocks_x, num_blocks_y), dim3(block_size_x, block_size_y)>>>(a.data_ptr<float>(), b.data_ptr<float>(), out.data_ptr<float>(), rows_a, cols_a, cols_b);

    out *= scaling_factor;
    return out;
}
"""

matmul_swish_cpp_source = (
    "torch::Tensor matmul_swish_cuda(torch::Tensor a, torch::Tensor b, float scaling_factor);"
)

# Compile the inline CUDA code for matrix multiplication and Swish activation
matmul_swish = load_inline(
    name="matmul_swish",
    cpp_sources=matmul_swish_cpp_source,
    cuda_sources=matmul_swish_source,
    functions=["matmul_swish_cuda"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)


class ModelNew(nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super(ModelNew, self).__init__()
        self.matmul_swish = matmul_swish

    def forward(self, x):
        return self.matmul_swish.matmul_swish_cuda(x, x, self.scaling_factor)


def get_inputs():
    batch_size = 128
    in_features = 32768
    return [torch.rand(batch_size, in_features)]


def get_init_inputs():
    in_features = 32768
    out_features = 32768
    scaling_factor = 2.0
    return [in_features, out_features, scaling_factor]
```
