import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for matrix multiplication and min operation
matrix_multiplication_min_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void matrix_multiplication_min_kernel(const float* a, const float* b, float* c, const float* d, float* e, int m, int n, int p) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < m && col < n) {
        float sum = 0.0f;
        for (int k = 0; k < p; ++k) {
            sum += a[row * p + k] * b[k * n + col];
        }

        float min_val = fminf(sum, d[col]);
        e[row * n + col] = min_val - d[col];
    }
}

torch::Tensor matrix_multiplication_min_cuda(torch::Tensor a, torch::Tensor b, torch::Tensor d, int m, int n, int p) {
    auto out = torch::zeros({m, n}, a.options());

    const int block_size = 256;
    dim3 grid((n + block_size - 1) / block_size, (m + block_size - 1) / block_size);
    dim3 block(block_size, block_size);

    matrix_multiplication_min_kernel<<<grid, block>>>(a.data_ptr<float>(), b.data_ptr<float>(), out.data_ptr<float>(), d.data_ptr<float>(), out.data_ptr<float>(), m, n, p);

    return out;
}
"""

matrix_multiplication_min_cpp_source = (
    "torch::Tensor matrix_multiplication_min_cuda(torch::Tensor a, torch::Tensor b, torch::Tensor d, int m, int n, int p);"
)

# Compile the inline CUDA code for matrix multiplication and min operation
matrix_multiplication_min = load_inline(
    name="matrix_multiplication_min",
    cpp_sources=matrix_multiplication_min_cpp_source,
    cuda_sources=matrix_multiplication_min_source,
    functions=["matrix_multiplication_min_cuda"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)

        x = self.matrix_multiplication_min.matrix_multiplication_min_cuda(x, self.linear.weight.t(), self.constant, x.size(0), x.size(1), self.linear.weight.size(1))
        return x

import torch
import torch.nn as nn

class Model(nn.Module):
    """
    Simple model that performs a matrix multiplication, applies minimum, and subtracts a constant.
    """
    def __init__(self, in_features, out_features, constant):
        super(Model, self).__init__()
        self.linear = nn.Linear(in_features, out_features)
        self.constant = nn.Parameter(torch.tensor(constant))

    def forward(self, x):
        x = self.linear(x)
        x = torch.min(x, self.constant)
        x = x - self.constant
        return x
