import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

bn_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void bn_kernel(const float* x, const float* mean, const float* var, float* y, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        y[idx] = (x[idx] - mean[0]) / sqrt(var[0] + 1e-5);
    }
}

torch::Tensor bn_cuda(torch::Tensor x, torch::Tensor mean, torch::Tensor var) {
    auto size = x.numel();
    auto y = torch::zeros_like(x);

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    bn_kernel<<<num_blocks, block_size>>>(x.data_ptr<float>(), mean.data_ptr<float>(), var.data_ptr<float>(), y.data_ptr<float>(), size);

    return y;
}
"""

bn_cpp_source = (
    "torch::Tensor bn_cuda(torch::Tensor x, torch::Tensor mean, torch::Tensor var);"
)

bn = load_inline(
    name="bn",
    cpp_sources=bn_cpp_source,
    cuda_sources=bn_source,
    functions=["bn_cuda"],
    verbose=True,
    extra_cflags=[""],
    extra_ldflags=[""],
)

class ModelNew(nn.Module):
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.scale = nn.Parameter(torch.randn(scale_shape))
        self.bn = bn

    def forward(self, x):
        x = self.gemm(x)
        x = x * self.scale
        x_mean = torch.mean(x, dim=0)
        x_var = torch.var(x, dim=0)
        x = self.bn.bn_cuda(x, x_mean, x_var)
        return x
