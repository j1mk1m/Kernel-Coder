__global__ void convolution_kernel(const float* input, const float* weight, const float* bias, float* output, int batch_size, int in_channels, int out_channels, int height, int width, int kernel_size, int stride, int padding) {
    int n = blockIdx.z * blockDim.z + threadIdx.z;
    if (n >= out_channels) return;

    int c = blockIdx.y * blockDim.y + threadIdx.y;
    if (c >= in_channels) return;

    int h_out = (height + 2 * padding - kernel_size) / stride + 1;
    int w_out = (width + 2 * padding - kernel_size) / stride + 1;

    int w_in = i % w_out + padding;

    float sum = bias[n];
    for (int k = 0; k < kernel_size; ++k) {

            }
        }
    }

    output[(n * batch_size + i) * in_channels + c] = sum;
}
