import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for fused linear and ReLU
fused_linear_relu_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void fused_linear_relu_kernel(const float* weight, const float* bias, const float* input, float* output, int batch_size, int in_features, int out_features) {
    int row_idx = blockIdx.y * blockDim.y + threadIdx.y;
    int col_idx = blockIdx.x * blockDim.x + threadIdx.x;

    if (row_idx < batch_size && col_idx < out_features) {
        float sum = bias[col_idx];
        for (int i = 0; i < in_features; ++i) {
            sum += weight[row_idx * in_features + i] * input[row_idx * in_features + i];
        }
        output[row_idx * out_features + col_idx] = max(sum, 0.0f);
    }
}

torch::Tensor fused_linear_relu_cuda(torch::Tensor weight, torch::Tensor bias, torch::Tensor input) {
    auto batch_size = input.size(0);
    auto in_features = input.size(1);
    auto out_features = weight.size(1);
    auto output = torch::zeros({batch_size, out_features}, input.options());

    const int block_size_x = 32;
    const int block_size_y = 32;
    const int num_blocks_x = (out_features + block_size_x - 1) / block_size_x;
    const int num_blocks_y = (batch_size + block_size_y - 1) / block_size_y;

    fused_linear_relu_kernel<<<dim3(num_blocks_x, num_blocks_y), dim3(block_size_x, block_size_y)>>>(
        weight.data_ptr<float>(), bias.data_ptr<float>(), input.data_ptr<float>(), output.data_ptr<float>(), batch_size, in_features, out_features);

    return output;
}
"""
