#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv_hswish_relu_kernel(const float* input, const float* weight, float* output, int batch_size, int channels_in, int channels_out, int height, int width, int kernel_size) {
    int n = blockIdx.x;
    int c = blockIdx.y;
    int h = blockIdx.z;
    int w = blockIdx.w;

    float sum = 0.0f;
    for (int kh = 0; kh < kernel_size; ++kh) {
        for (int kw = 0; kw < kernel_size; ++kw) {
            int ih = h + kh;
            int iw = w + kw;
            if (ih >= 0 && ih < height && iw >= 0 && iw < width) {
                sum += input[n * channels_in * height * width + (c * kernel_size * kernel_size + kh * kernel_size + kw) * height * width + ih * width + iw] * weight[c * kernel_size * kernel_size + kh * kernel_size + kw];
            }
        }
    }

    output[n * channels_out * height * width + c * height * width + h * width + w] = std::max(std::min(sum * 6.0f + 3.0f, 6.0f), 0.0f);
}

torch::Tensor conv_hswish_relu_cuda(torch::Tensor input, torch::Tensor weight) {
    int batch_size = input.size(0);
    int channels_in = input.size(1);
    int channels_out = weight.size(0);
    int height = input.size(2);
    int width = input.size(3);
    int kernel_size = weight.size(2);

    auto output = torch::zeros({batch_size, channels_out, height, width}, input.options());

    dim3 threads_per_block(16, 16, 1);
    dim3 blocks_per_grid((width + threads_per_block.x - 1) / threads_per_block.x,
                          (height + threads_per_block.y - 1) / threads_per_block.y,
                          (channels_out + threads_per_block.z - 1) / threads_per_block.z);

    conv_hswish_relu_kernel<<<blocks_per_grid, threads_per_block>>>(input.data_ptr<float>(), weight.data_ptr<float>(), output.data_ptr<float>(), batch_size, channels_in, channels_out, height, width, kernel_size);

    return output;
}
